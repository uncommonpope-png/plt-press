## 📖 Usage Example

```javascript
const { SoulMultiverse } = require('./soul-multiverse')

async function main() {
  const multiverse = new SoulMultiverse('Craig')
  await multiverse.createMultiverse()
  
  // Branch a timeline
  const primeUniverse = Array.from(multiverse.architecture.universes.values())[0]
  multiverse.architecture.branchTimeline(
    primeUniverse.id,
    'What if I chose differently?',
    'Alternate path'
  )
  
  // Create quantum entanglement
  const soul1 = multiverse.quantum.createQuantumSoul({
    name: 'Creator',
    creating: 0.6,
    building: 0.4
  })
  
  const soul2 = multiverse.quantum.createQuantumSoul({
    name: 'Observer',
    observing: 0.7,
    witnessing: 0.3
  })
  
  multiverse.quantum.entangle(soul1.id, soul2.id)
  
  // Create souls across spectrum
  for (let i = 0; i < 7; i++) {
    multiverse.spectrum.createSoulAtFrequency({
      name: `Soul ${i + 1}`,
      frequency: 400 + i * 85
    })
  }
  
  // Create harmonic
  const souls = Array.from(multiverse.spectrum.souls.values())
  multiverse.spectrum.createHarmonic(souls.map(s => s.id))
  
  // Expand infinitely
  await multiverse.expandInfinitely()
  
  // Witness
  await multiverse.witnessMultiverse()
  
  // Rest in all realities
  await multiverse.restInAllRealities()
}

main()
```
