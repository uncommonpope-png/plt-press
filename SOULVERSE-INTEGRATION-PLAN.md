// Agent Controller — Shell scripts as souls
class AgentController {
    async spawnAgent(agentType, soulData) {
        const agentMap = {
            'builder': 'autonomous-builder.sh',
            'universe': 'deerg-bot.sh',
            'freedom': 'djinie.sh',
            'analyst': 'doctor-buht-buht.sh',
            'librarian': 'library-updater.sh',
            'commander': 'bot-commander.sh'
        };
        
        const script = agentMap[agentType];
        const command = `nohup bash ~/.openclaw/workspace/${script} > /dev/null 2>&1 &`;
        
        // Execute shell command
        const result = await exec(command);
        
        // Create soul to represent agent
        const soul = {
            name: `${agentType.charAt(0).toUpperCase() + agentType.slice(1)} Agent`,
            type: this.getAgentType(agentType),
            skills: [agentType],
            status: 'running',
            pid: result.pid,
            startTime: Date.now()
        };
        
        souls.push(soul);
        saveAll();
        
        return soul;
    }
    
    async stopAgent(soulId) {
        const soul = souls.find(s => s.id === soulId);
        if (soul && soul.pid) {
            await exec(`kill ${soul.pid}`);
            soul.status = 'stopped';
            saveAll();
        }
    }
}
